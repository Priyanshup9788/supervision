#include <stdio.h>

struct FeedbackForm{
    int  grid ;
    char name[100] ;
    char course[100] ;
    char branchName[100] ;
    float grade ;
}s1,s2,s3;

int main() {
    
    struct FeedbackForm arti[3] ;
    
    s1.grid = 3424;
    printf("%d",s1.grid);
    s2.grid = 3424;
    printf("%d",s1.grid);
    
    
    for(int i = 0 ; i < 3 ; i++)
    {
        printf("Enter grid : ");
        scanf("%d",&arti[i].grid);
        printf("Enter name : ");
        scanf("%s",arti[i].name);
        printf("Enter course : ");
        scanf("%s",arti[i].course);
        printf("Enter branchName : ");
        scanf("%s",arti[i].branchName);
        printf("Enter grade : ");
        scanf("%f",&arti[i].grade);
